def electro_strain(V, t, M):
return M * (V/t)**2

def spatial_electro_strain(x, L, V, t, M):
return M * (V/t)**2 * np.sin(np.pi * x / L)

## physics/plasticity.py

def von_mises(stress_tensor):
s1, s2, s3 = stress_tensor
return np.sqrt(0.5*((s1-s2)**2 + (s2-s3)**2 + (s3-s1)**2))

def check_yield(sigma_vm, sigma_y):
return sigma_vm >= sigma_y
